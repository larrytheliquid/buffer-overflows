def octets
  addr = self.to_i
  [ 
    ((addr >> 24) & 0xff),
    ((addr >> 16) & 0xff),
    ((addr >> 8) & 0xff),
    (addr & 0xff)
  ]
end

          def read_quad(str)
                  match = IPV4_RE.match(str)
                  if match.nil?
                          raise ArgumentError.new("str is not a valid IPV4 address")
                  end 
                          a = match[1].to_i
                          b = match[2].to_i
                          c = match[3].to_i
                          d = match[4].to_i
                  unless (a >= 0 && a <= 255 &&
                                                  b >= 0 && b <= 255 &&
                                                  c >= 0 && c <= 255 &&
                                                  d >= 0 && d <= 255)
                          raise ArgumentError.new("str is not a valid IPV4 address")
                  end

self[:ip_addr].value = (a<<24) + (b<<16) + (c<<8) + d
                  self
          end
