def read(str)
        force_binary(str)
        return self if str.nil? || str.size.zero?
        if 1.respond_to? :ord
                self[:hlen] = (str[0].ord & 0b11110000) >> 4
        else
                self[:hlen] = (str[0] & 0b11110000) >> 4
        end
        self
end
